73/* Function: rtIOStreamOpen */



74int rtIOStreamOpen(int argc, void * argv[])



75{



76    (void)argc;



77    (void)argv;



78    



79    /* Initialize receive structure */



80    UART_Rx.buffer   = &UART_rxBuffer[0];



81    UART_Rx.position = 0;



82    



83    /* Initialize transmit structure */



84    UART_Tx.buffer   = &UART_txBuffer[0];



85    UART_Tx.r        = 0;



86    UART_Tx.w        = 0;



87    UART_Tx.dmaState = MW_DMA_STATE_IDLE;  



88    



89    /* Enable DMA / GPIO clocks */



90    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_DMA1 | RCC_AHB1Periph_GPIOA, ENABLE);



91    /* Enable USART clock */



92    RCC_APB1PeriphClockCmd(RCC_APB1Periph_USART2, ENABLE);



93    



94    /* Setup DMA's first and then start USART. Note USART must be initialize / started



95       after DMA's to avoid race conditions. The DMA enable request for UART is



96       enabled at the end of USART2_init */



97    DMA_RxInit();



98    DMA_TxInit();



99    USART2_init(4 * 115200);



100 



101    return SERVER_STREAM_ID;



102}

112/* Function: rtIOStreamSend */



113int rtIOStreamSend2(int streamID, const void *src, size_t size, size_t *sizeSent)



114{



115    u16 i = 0;



116    char* charToSend = (char *)src;



117    (void)streamID; 



118    



119    for (i=0; i<size; i++) {



120        while (USART_GetFlagStatus(USART2, USART_FLAG_TXE) == RESET) ;



121        USART_SendData(USART2, *charToSend++);



122    }



123    *sizeSent = i;



124    



125    return RTIOSTREAM_NO_ERROR;



126}



127



128/* Function: rtIOStreamSend */



129int rtIOStreamSend(int streamID, const void *src, size_t size, size_t *sizeSent)



130{



131    (void)streamID; 



132    



133    /* Wait until DMA engine is idle before scheduling a new one */



134    while (DMA_GetCmdStatus(DMA1_Stream6) != DISABLE);



135    size = size < UART_TX_BUFFER_SIZE ? size : UART_TX_BUFFER_SIZE;



136    memcpy(UART_Tx.buffer, src, size);



137    



138    /* Configure DMA engines */



139    UART_Tx.DMA_Config.DMA_BufferSize      = (uint16_t)size;



140    UART_Tx.DMA_Config.DMA_Memory0BaseAddr = (uint32_t)&(UART_Tx.buffer[0]);



141    DMA_Init(DMA1_Stream6, &(UART_Tx.DMA_Config));



142    DMA_ITConfig(DMA1_Stream6, DMA_IT_TC, ENABLE);



143    DMA_Cmd(DMA1_Stream6, ENABLE);



144    



145    /* Return number of bytes sent */



146    *sizeSent = size;



147    



148    return RTIOSTREAM_NO_ERROR;



149}

152/* TX DMA IRQ handler */



153void DMA1_Stream6_IRQHandler(void)



154{



155    /* Test on DMA Stream Transfer Complete interrupt */



156    if (DMA_GetITStatus(DMA1_Stream6, DMA_IT_TCIF6))



157    {



158        /* Clear DMA Stream Transfer Complete interrupt pending bit */



159        DMA_ClearITPendingBit(DMA1_Stream6, DMA_IT_TCIF6);



160    }



161}

163/* Function: rtIOStreamRecv */



164int rtIOStreamRecv(int streamID, void *dst, size_t size, size_t *sizeRecv)



165{



166    (void)streamID; 



167    



168    /* Pass available data in the UART buffer to application */



169    if (UART_getRxBytes() >= size) {



170        int i;



171        u8 *srcPtr, *dstPtr;



172        u16 pos;



173        



174        pos = UART_Rx.position;



175        srcPtr = UART_Rx.buffer;



176        dstPtr = (u8 *) dst;



177        for (i = 0; i < size; i++) {



178            *dstPtr++ = srcPtr[pos];



179            pos = (pos + 1) & (UART_RX_BUFFER_SIZE - 1);



180        };



181        UART_Rx.position = (UART_Rx.position + size) & (UART_RX_BUFFER_SIZE - 1);



182        *sizeRecv = size;



183    }



184    else {



185        *sizeRecv = 0;



186    }



187    



188    return RTIOSTREAM_NO_ERROR;



189}

191/* Initialize RX DMA */



192static void DMA_RxInit(void)



193{



194    NVIC_InitTypeDef NVIC_InitStructure;



195    DMA_InitTypeDef  DMA_InitStructure;



196    s32 timeout;



197    



198    /* USART2 DMA mapping:



199     * P:216/1416



200     * USART2_RX: DMA1/Stream5/Channel4



201     * USART2_TX: DMA1/Stream6/Channel4



202     */



203    DMA_DeInit(DMA1_Stream5);



204    while (DMA_GetCmdStatus(DMA1_Stream5) != DISABLE);



205    



206    /* Configure DMA stream */



207    DMA_InitStructure.DMA_Channel            = DMA_Channel_4;



208    DMA_InitStructure.DMA_DIR                = DMA_DIR_PeripheralToMemory;



209    DMA_InitStructure.DMA_Memory0BaseAddr    = (uint32_t)UART_Rx.buffer;



210    DMA_InitStructure.DMA_BufferSize         = (uint16_t)UART_RX_BUFFER_SIZE;



211    DMA_InitStructure.DMA_PeripheralBaseAddr = (uint32_t)(&(USART2->DR));



212    DMA_InitStructure.DMA_PeripheralInc      = DMA_PeripheralInc_Disable;



213    DMA_InitStructure.DMA_MemoryInc          = DMA_MemoryInc_Enable;



214    DMA_InitStructure.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Byte;



215    DMA_InitStructure.DMA_MemoryDataSize     = DMA_MemoryDataSize_Byte;



216    DMA_InitStructure.DMA_Mode               = DMA_Mode_Circular;



217    DMA_InitStructure.DMA_Priority           = DMA_Priority_VeryHigh;



218    DMA_InitStructure.DMA_FIFOMode           = DMA_FIFOMode_Disable;



219    DMA_InitStructure.DMA_FIFOThreshold      = DMA_FIFOThreshold_Full;



220    DMA_InitStructure.DMA_MemoryBurst        = DMA_MemoryBurst_Single;



221    DMA_InitStructure.DMA_PeripheralBurst    = DMA_PeripheralBurst_Single;



222    DMA_Init(DMA1_Stream5, &DMA_InitStructure);



223    



224    /* Initialize DMA interrupt */



225    NVIC_InitStructure.NVIC_IRQChannel                   = DMA1_Stream5_IRQn;



226    NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0x0;



227    NVIC_InitStructure.NVIC_IRQChannelSubPriority        = 0x0;



228    NVIC_InitStructure.NVIC_IRQChannelCmd                = ENABLE;



229    NVIC_Init(&NVIC_InitStructure);



230    



231    /* Enable DMA Stream Transfer Complete interrupt */



232    DMA_ITConfig(DMA1_Stream5, DMA_IT_TC, ENABLE);



233    



234    /* Enable the DMA RX Stream */



235    DMA_Cmd(DMA1_Stream5, ENABLE);



236    



237    /* Wait until DMA is enabled before */



238    timeout = 1000000;



239    while ((DMA_GetCmdStatus(DMA1_Stream5) != ENABLE) && (timeout-- > 0));



240    if (timeout <= 0) {



241        MW_USART_ERRNO = MW_USART_RX_INIT_TIMEOUT;



242        while (1);



243    }



244}

246/* Initialize TX DMA */



247static void DMA_TxInit(void)



248{



249    NVIC_InitTypeDef NVIC_InitStructure;



250    



251    /* USART2 DMA mapping:



252     * P:216/1416



253     * USART2_RX: DMA1/Stream5/Channel4



254     * USART2_TX: DMA1/Stream6/Channel4



255     */



256    DMA_DeInit(DMA1_Stream6);



257    while (DMA_GetCmdStatus(DMA1_Stream6) != DISABLE);



258    



259    /* Configure DMA stream */



260    UART_Tx.DMA_Config.DMA_Channel            = DMA_Channel_4;



261    UART_Tx.DMA_Config.DMA_DIR                 = DMA_DIR_MemoryToPeripheral;



262    UART_Tx.DMA_Config.DMA_Memory0BaseAddr    = (uint32_t)UART_Tx.buffer;



263    UART_Tx.DMA_Config.DMA_BufferSize         = (uint16_t)1;



264    UART_Tx.DMA_Config.DMA_PeripheralBaseAddr = (uint32_t)(&(USART2->DR));



265    UART_Tx.DMA_Config.DMA_PeripheralInc      = DMA_PeripheralInc_Disable;



266    UART_Tx.DMA_Config.DMA_MemoryInc          = DMA_MemoryInc_Enable;



267    UART_Tx.DMA_Config.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Byte;



268    UART_Tx.DMA_Config.DMA_MemoryDataSize     = DMA_MemoryDataSize_Byte;



269    UART_Tx.DMA_Config.DMA_Mode               = DMA_Mode_Normal;



270    UART_Tx.DMA_Config.DMA_Priority           = DMA_Priority_VeryHigh;



271    UART_Tx.DMA_Config.DMA_FIFOMode           = DMA_FIFOMode_Disable;



272    UART_Tx.DMA_Config.DMA_FIFOThreshold      = DMA_FIFOThreshold_Full;



273    UART_Tx.DMA_Config.DMA_MemoryBurst        = DMA_MemoryBurst_Single;



274    UART_Tx.DMA_Config.DMA_PeripheralBurst    = DMA_PeripheralBurst_Single;



275    



276    /* Initialize DMA interrupt */



277    NVIC_InitStructure.NVIC_IRQChannel                   = DMA1_Stream6_IRQn;



278    NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0x0;



279    NVIC_InitStructure.NVIC_IRQChannelSubPriority        = 0x0;



280    NVIC_InitStructure.NVIC_IRQChannelCmd                = ENABLE;



281    NVIC_Init(&NVIC_InitStructure);



282}

284/* Available data */



285static s32 UART_getRxBytes(void)



286{



287    s32 dmaPosition;



288    



289    dmaPosition = UART_RX_BUFFER_SIZE - DMA_GetCurrDataCounter(DMA1_Stream5);



290    if (dmaPosition >= UART_Rx.position) {



291        return(dmaPosition - UART_Rx.position);



292    }



293    else {



294        /* DMA pointer has wrapped around */



295        return(dmaPosition + (UART_RX_BUFFER_SIZE - UART_Rx.position));



296    }



297}

311/* Function: USART2_init */



312static void USART2_init(unsigned int baudRate)



313{



314    GPIO_InitTypeDef GPIO_config;



315    USART_InitTypeDef USART_config;



316    



317    /* Configure GPIO */



318    GPIO_StructInit(&GPIO_config);



319    GPIO_config.GPIO_Pin   = GPIO_Pin_2 | GPIO_Pin_3;



320    GPIO_config.GPIO_Mode  = GPIO_Mode_AF;



321    GPIO_config.GPIO_OType = GPIO_OType_PP;



322    GPIO_config.GPIO_PuPd  = GPIO_PuPd_UP;



323    GPIO_config.GPIO_Speed = GPIO_Speed_50MHz;



324    GPIO_Init(GPIOA, &GPIO_config);



325    



326    /* Connect USART pins to AF */



327    GPIO_PinAFConfig(GPIOA, GPIO_PinSource2, GPIO_AF_USART2);



328    GPIO_PinAFConfig(GPIOA, GPIO_PinSource3, GPIO_AF_USART2);



329    



330    /* Set USART parameters */



331    USART_StructInit(&USART_config);



332    USART_config.USART_BaudRate            = baudRate;



333    USART_config.USART_WordLength          = USART_WordLength_8b;



334    USART_config.USART_StopBits            = USART_StopBits_1;



335    USART_config.USART_Parity              = USART_Parity_No ;



336    USART_config.USART_HardwareFlowControl = USART_HardwareFlowControl_None;



337    USART_config.USART_Mode                = USART_Mode_Rx | USART_Mode_Tx ;



338    



339    /* Configure and enable USART2 */



340    USART_Init(USART2, &USART_config);



341    



342    /* Enable the USART RX-TX DMA requests */



343    USART_DMACmd(USART2, USART_DMAReq_Rx, ENABLE);



344    USART_DMACmd(USART2, USART_DMAReq_Tx, ENABLE);



345    USART_Cmd(USART2, ENABLE);



346}
