7static GPIO_TypeDef* GPIO_base_array[9] = {GPIOA, GPIOB, GPIOC, GPIOD, GPIOE, GPIOF, GPIOG, GPIOH, GPIOI};



8



9/*



10 *******************************************



11 * GPIO Wrapper functions



12 *******************************************



13 */



14uint32_t GpioClockEnableBit(uint8_t GPIOx_selected)



15{



16    return ((uint32_t) (1U << (uint32_t)(GPIOx_selected)));



17}



18



19void MW_GpioInit(uint8_t GPIOx_selected, 



20                 uint32_t GPIO_Pin, 



21                 uint8_t GPIO_Mode, 



22                 uint8_t GPIO_Speed, 



23                 uint8_t GPIO_OType, 



24                 uint8_t GPIO_PuPd)



25{



26    GPIO_InitTypeDef GPIO_InitStruct;



27



28	/* Enable GPIO clock */



29    RCC_AHB1PeriphClockCmd(GpioClockEnableBit(GPIOx_selected), ENABLE);



30    



31    GPIO_InitStruct.GPIO_Pin = GPIO_Pin;



32    GPIO_InitStruct.GPIO_Mode = (GPIOMode_TypeDef)GPIO_Mode;



33    GPIO_InitStruct.GPIO_Speed = (GPIOSpeed_TypeDef)GPIO_Speed;



34    GPIO_InitStruct.GPIO_OType = (GPIOOType_TypeDef)GPIO_OType;



35    GPIO_InitStruct.GPIO_PuPd = (GPIOPuPd_TypeDef)GPIO_PuPd;



36    GPIO_Init(GPIO_base_array[GPIOx_selected], &GPIO_InitStruct);



37}



38



39void MW_RCC_AHB1PeriphClockCmd(uint8_t GPIOx_selection, uint8_t enable_or_disable)



40{



41    RCC_AHB1PeriphClockCmd(GpioClockEnableBit(GPIOx_selection), ENABLE);



42}



43



44uint8_t MW_GPIO_ReadInputDataBit(uint8_t GPIOx_selected, uint8_t GPIO_Pin)



45{



46    return GPIO_ReadInputDataBit(GPIO_base_array[GPIOx_selected], 



47                                  MW_GetGpioPinPosition(GPIO_Pin));



48}



49



50void MW_GPIO_WriteBit(uint8_t GPIOx_selected, uint16_t GPIO_Pin, uint8_t BitVal)



51{



52    GPIO_WriteBit(GPIO_base_array[GPIOx_selected], MW_GetGpioPinPosition(GPIO_Pin), (BitAction)BitVal);



53}



54



55void MW_GPIO_ToggleBits(uint8_t GPIOx_selected, uint16_t GPIO_Pin, uint8_t BitVal)



56{



57	/* Toggle pin when input is high */



58    if (BitVal)



59	{



60    GPIO_ToggleBits(GPIO_base_array[GPIOx_selected], MW_GetGpioPinPosition(GPIO_Pin));



61}



62}
