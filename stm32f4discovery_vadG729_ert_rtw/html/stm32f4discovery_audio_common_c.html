8/* I2S_Clock_config: Configuring the prescalers to generate required I2S clock */



9void MW_I2S_Clock_config(uint32_t plln, uint32_t pllr)



10{



11    /* Clock selection to I2S */



12    /* Disable PLLI2S */



13    RCC_PLLI2SCmd(DISABLE);



14    /* Select PLLI2S as the clock source */



15    RCC_I2SCLKConfig(RCC_I2S2CLKSource_PLLI2S);



16    



17    /* Generate the I2SCLK set by PLLN and PLLR */



18    RCC->PLLI2SCFGR = (pllr << 28) | (plln << 6);



19    



20    /* Enable PLLI2S and wait until it is ready */



21	RCC_PLLI2SCmd(ENABLE);



22	while (!(RCC ->CR & RCC_CR_PLLI2SRDY));



23}

25void MW_I2S_GetPrescaler(uint32_t audio_sample_freqHz, 



26                         uint8_t num_chls, 



27                         uint8_t nb_bits_per_sample, 



28                         uint8_t enable_master_clk, 



29                         uint8_t * i2sdiv, 



30                         uint8_t * i2sodd)



31{



32    float prescaler_term_fl;



33    uint32_t i2s_clk;



34    uint32_t prescaler_term;



35    uint16_t i2splln, i2spllr, pllm;



36    uint16_t i2sdiv_tmp;



37    uint8_t i2sodd_tmp;



38    



39    



40    /* Force PLLI2S as I2S clock source */



41    if ((RCC->CFGR & RCC_CFGR_I2SSRC) != 0)



42    {



43      RCC->CFGR &= ~(uint32_t)RCC_CFGR_I2SSRC;



44    }    



45    



46    /* PLLI2SN */



47    i2splln = (uint32_t)(((RCC->PLLI2SCFGR & RCC_PLLI2SCFGR_PLLI2SN) >> 6) & \



48                      (RCC_PLLI2SCFGR_PLLI2SN >> 6));



49    



50    /* PLLI2SR */



51    i2spllr = (uint32_t)(((RCC->PLLI2SCFGR & RCC_PLLI2SCFGR_PLLI2SR) >> 28) & \



52                      (RCC_PLLI2SCFGR_PLLI2SR >> 28));



53    



54    /* PLLM */



55    pllm = (uint32_t)(RCC->PLLCFGR & RCC_PLLCFGR_PLLM);      



56    



57    /* Calculate I2S source clock */



58    i2s_clk = (uint32_t)(((HSE_VALUE / pllm) * i2splln) / i2spllr);



59    



60    /* Prescaler term */



61    prescaler_term_fl = (float)i2s_clk / (float)(nb_bits_per_sample * num_chls * audio_sample_freqHz);



62    if (enable_master_clk)



63    {



64        assert_param((16 == nb_bits_per_sample) || (32 == nb_bits_per_sample));



65        prescaler_term_fl /= (float)((16 == nb_bits_per_sample) ? 8 : 4);



66    }



67    /* Round */



68    prescaler_term = (uint32_t)(prescaler_term_fl + 0.5);



69    



70    /* Check the parity of the divider */



71    i2sodd_tmp = (uint16_t)(prescaler_term & (uint16_t)0x0001);



72   



73    /* Compute the i2sdiv prescaler */



74    i2sdiv_tmp = (uint16_t)((prescaler_term - i2sodd_tmp) / 2);



75    



76    if (i2sdiv_tmp < 2)



77    {



78        //Default values



79        *i2sdiv = 2;



80        *i2sodd = 0;



81    }



82    else if (i2sdiv_tmp > 255)



83    {



84        *i2sdiv = 255;



85        *i2sodd = 0;



86    }



87    else



88    {



89        *i2sdiv = (uint8_t)i2sdiv_tmp;



90        *i2sodd = i2sodd_tmp;



91    }



92}

94void MW_I2S_SetPrescaler(SPI_TypeDef* SPIx, uint8_t i2sdiv, uint8_t i2sodd, I2S_InitTypeDef* I2S_InitStruct)



95{



96    uint16_t tmpreg = 0;



97    



98    /* Get I2S Configuration */



99    tmpreg = SPIx->I2SCFGR;



100    



101    /* Write to SPIx I2SPR register the computed value */



102    SPIx->I2SPR = (uint16_t)((uint16_t)i2sdiv | (uint16_t)((i2sodd << 8) | (uint16_t)I2S_InitStruct->I2S_MCLKOutput));



103    



104    /* Configure the I2S with the SPI_InitStruct values */



105    tmpreg |= (uint16_t)((uint16_t)SPI_I2SCFGR_I2SMOD | (uint16_t)(I2S_InitStruct->I2S_Mode | \



106                    (uint16_t)(I2S_InitStruct->I2S_Standard | (uint16_t)(I2S_InitStruct->I2S_DataFormat | \



107                    (uint16_t)I2S_InitStruct->I2S_CPOL))));



108    



109    /* Write to SPIx I2SCFGR */  



110    SPIx->I2SCFGR = tmpreg;



111}
